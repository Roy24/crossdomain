window.addEventListener("load", gTracking, false);
function gTracking() {
	jQuery( '#ct' ).one( 'click', function() {
		window.google_trackConversion({
		      google_conversion_id: 123456789,
		      google_custom_params: {
			parameter1: 'abc123',
			parameter2: 29.99
		      },
		      google_remarketing_only: true
		    });
	});
}
